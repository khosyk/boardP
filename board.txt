CREATE TABLE contents (
`id` INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
`userId` VARCHAR(50),
`title` VARCHAR(50),
`body` VARCHAR(9999),
`img` VARCHAR(500),
`like` INT DEFAULT '0',
`share` INT DEFAULT '0',
`likeActive` tinyint(1) DEFAULT '0'  ,
`shareActive` tinyint(1) DEFAULT '0' ,
`replies` VARCHAR(500)
);
